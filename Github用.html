    <script>

        // ==========================================

        // 設定エリア (ここを書き換えてください)

        // ==========================================

        

        // 1. LINE Developersコンソールで取得したLIFF ID

        const MY_LIFF_ID = "2009010005-liWeUjh7"; 


        // 2. Google Apps ScriptのウェブアプリURL

        const GAS_API_URL = "https://script.google.com/macros/s/AKfycbylj4WLoiBDIs38hzbDm0R0fvuPiLk-G5Nxc9lkqqfxhLtaRYtRLOq1kXW04i0oncEe/exec";


        // 3. ショップカードのURL (LINE公式アカウントの管理画面から取得)

        // 例: https://lin.ee/xxxxxxx

        const SHOP_CARD_URL = "https://u.lin.ee/6R7EcIP";


        // ==========================================

        // アプリケーションロジック

        // ==========================================


        const loadingEl = document.getElementById('loading');

        const loadingSpinnerEl = document.getElementById('loading-spinner');

        const loadingTextEl = document.getElementById('loading-text');

        const formContainerEl = document.getElementById('form-container');

        const completeContainerEl = document.getElementById('complete-container');

        const userPictureEl = document.getElementById('userPicture');

        const lineDisplayNameEl = document.getElementById('lineDisplayName');

        const shopCardLinkEl = document.getElementById('shopCardLink');

        const submitBtn = document.getElementById('submitBtn');


        let lineProfile = null;


        // 初期化処理

        window.onload = async function() {

            // エラー回避: LIFF IDが初期値のままの場合は処理を止めて警告を表示

            if (MY_LIFF_ID === "2009010005-liWeUjh7") {

                if(loadingSpinnerEl) loadingSpinnerEl.style.display = 'none';

                loadingTextEl.innerHTML = '<span class="text-red-500 font-bold text-base block mb-2">⚠️ 準備が完了していません</span>コード内の <code>MY_LIFF_ID</code> を、LINE Developersコンソールで取得した実際のLIFF IDに書き換えてください。';

                return;

            }


            try {

                // LIFF初期化

                await liff.init({ liffId: MY_LIFF_ID });


                if (!liff.isLoggedIn()) {

                    liff.login();

                    return;

                }


                // プロフィール取得

                lineProfile = await liff.getProfile();

                

                // 画面への反映

                lineDisplayNameEl.textContent = lineProfile.displayName;

                if (lineProfile.pictureUrl) {

                    userPictureEl.src = lineProfile.pictureUrl;

                    userPictureEl.classList.remove('hidden');

                }


                // ローディング終了、フォーム表示

                loadingEl.classList.add('hidden');

                formContainerEl.classList.remove('hidden');


            } catch (err) {

                console.error('LIFF Init Error', err);

                if(loadingSpinnerEl) loadingSpinnerEl.style.display = 'none';

                loadingTextEl.innerHTML = '<span class="text-red-500">LINEとの連携に失敗しました。</span><br><span class="text-xs text-gray-400 mt-2 block">エラー: ' + err.message + '</span>';

            }

        };


        // 送信処理

        document.getElementById('registrationForm').addEventListener('submit', async function(e) {

            e.preventDefault();

            

            if (!lineProfile) return;


            const realName = document.getElementById('realName').value;

            

            // ボタンをローディング状態に

            const originalBtnText = submitBtn.innerHTML;

            submitBtn.disabled = true;

            submitBtn.innerHTML = '<svg class="animate-spin -ml-1 mr-3 h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg> 処理中...';


            try {

                const data = {

                    userId: lineProfile.userId,

                    displayName: lineProfile.displayName,

                    realName: realName

                };


                await fetch(GAS_API_URL, {

                    method: 'POST',

                    mode: 'no-cors', // 重要: GASへの直接POST用

                    headers: {

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify(data)

                });


                // 完了画面へ遷移

                formContainerEl.classList.add('hidden');

                completeContainerEl.classList.remove('hidden');

                

                // ショップカードのリンクを設定

                shopCardLinkEl.href = SHOP_CARD_URL;


            } catch (error) {

                console.error('Submit Error', error);

                alert('登録に失敗しました。もう一度お試しください。');

                submitBtn.disabled = false;

                submitBtn.innerHTML = originalBtnText;

            }

        });

    </script>
