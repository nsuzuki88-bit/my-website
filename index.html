        <!-- ヘッダー -->

        <div class="text-center mb-8">

            <h1 class="text-2xl font-bold text-gray-800">ご登録をお願いします</h1>

            <p class="text-sm text-gray-500 mt-2">登録完了後、ショップカードを発行します</p>

        </div>


        <!-- ローディング表示 -->

        <div id="loading" class="text-center py-4">

            <div class="animate-spin rounded-full h-8 w-8 border-b-2 border-green-500 mx-auto"></div>

            <p class="text-sm text-gray-500 mt-2">システムを準備中です...</p>

        </div>


        <!-- ログインプロンプト (新規追加) -->

        <div id="loginPrompt" class="hidden text-center space-y-4 mt-4">

            <p class="text-gray-600 text-sm">システムを利用するにはLINE連携が必要です。</p>

            <button id="loginBtn" class="w-full flex justify-center py-3 px-4 border border-transparent rounded-md shadow-sm text-sm font-medium text-white bg-[#06C755] hover:bg-[#04a646] focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-[#06C755] transition-colors">

                LINEでログインする

            </button>

        </div>


        <!-- 登録フォーム (初期状態は非表示) -->

        <form id="registerForm" class="hidden space-y-6">

            <div>

                <label for="name" class="block text-sm font-medium text-gray-700">お名前（フルネーム）</label>

                <input type="text" id="name" name="name" required

                    class="mt-1 block w-full px-4 py-3 border border-gray-300 rounded-md shadow-sm focus:outline-none focus:ring-green-500 focus:border-green-500"

                    placeholder="例: 山田 太郎">

            </div>


            <button type="submit" id="submitBtn"

                class="w-full flex justify-center py-3 px-4 border border-transparent rounded-md shadow-sm text-sm font-medium text-white bg-green-600 hover:bg-green-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-green-500 transition-colors">

                登録してショップカードを見る

            </button>

        </form>


        <!-- エラーメッセージ表示エリア -->

        <div id="errorMessage" class="hidden mt-4 p-3 bg-red-50 text-red-700 text-sm rounded-md text-center">

            <!-- エラー内容がここに表示されます -->

        </div>

    </div>


    <script>

        // 設定値の定義

        const LIFF_ID = "2009010005-liWeUjh7";

        const GAS_API_URL = "https://script.google.com/macros/s/AKfycbyf7y50-xIAAXyiGlUf1Yz5F8erpXVXl_y2EmKUpFsdd4akn6NFos4halVeYamwdDcp/exec";

        const SHOP_CARD_URL = "https://u.lin.ee/6R7EcIP";


        // DOM要素の取得

        const loading = document.getElementById('loading');

        const loginPrompt = document.getElementById('loginPrompt');

        const loginBtn = document.getElementById('loginBtn');

        const form = document.getElementById('registerForm');

        const submitBtn = document.getElementById('submitBtn');

        const errorMessage = document.getElementById('errorMessage');

        let lineProfile = null;


        // LIFFの初期化

        async function initializeLiff() {

            try {

                await liff.init({ liffId: LIFF_ID });

                

                if (liff.isLoggedIn()) {

                    // プロフィール情報の取得 (LINE UID, 表示名など)

                    lineProfile = await liff.getProfile();

                    // UIの切り替え

                    loading.classList.add('hidden');

                    form.classList.remove('hidden');

                } else {

                    // 自動リダイレクトをやめて、ログインボタンを表示する

                    loading.classList.add('hidden');

                    loginPrompt.classList.remove('hidden');

                }

            } catch (err) {

                showError("LINE連携の初期化に失敗しました。LINEアプリから再度お試しください。");

                console.error('LIFF Initialization failed ', err);

            }

        }


        // ログインボタンの処理

        loginBtn.addEventListener('click', () => {

            liff.login();

        });


        // フォーム送信時の処理

        form.addEventListener('submit', async (e) => {

            e.preventDefault();

            const nameInput = document.getElementById('name').value;


            // 送信ボタンをローディング状態にする

            submitBtn.disabled = true;

            submitBtn.innerHTML = '<div class="animate-spin rounded-full h-5 w-5 border-b-2 border-white mr-2"></div> 登録中...';

            submitBtn.classList.add('opacity-75', 'cursor-not-allowed');

            errorMessage.classList.add('hidden');


            // GASへ送信するデータ

            const postData = {

                uid: lineProfile.userId,

                displayName: lineProfile.displayName,

                name: nameInput

            };


            try {

                // GASへのリクエスト送信 (CORS対策のため text/plain で送信)

                const response = await fetch(GAS_API_URL, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'text/plain', 

                    },

                    body: JSON.stringify(postData)

                });


                const result = await response.json();


                if (result.status === "success") {

                    // 登録成功時：ショップカードへ遷移

                    window.location.href = SHOP_CARD_URL;

                } else {

                    throw new Error("サーバー側でエラーが発生しました。");

                }

            } catch (err) {

                showError("登録に失敗しました。電波状況の良い場所で再度お試しください。");

                console.error('Post error:', err);

                

                // ボタンの状態を元に戻す

                submitBtn.disabled = false;

                submitBtn.innerHTML = '登録してショップカードを見る';

                submitBtn.classList.remove('opacity-75', 'cursor-not-allowed');

            }

        });


        // エラーメッセージ表示関数

        function showError(msg) {

            loading.classList.add('hidden');

            errorMessage.textContent = msg;

            errorMessage.classList.remove('hidden');

        }


        // ページ読み込み時にLIFFを初期化

        window.addEventListener('load', initializeLiff);

    </script>
